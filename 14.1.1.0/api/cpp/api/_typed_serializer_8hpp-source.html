00024 /**
00025 * A PofSerializer implementation that supports the serialization and
00026 * deserialization of any class to and from a POF stream, by delegating the
00027 * serialization logic to templated free functions.
00028 *
00029 * The following set of free functions must be defined for the templated type:
00030 * <ul>
00031 *   <li>template<> void serialize<T>(PofWriter::Handle, const T&)</li>
00032 *   <li>template<> T deserialize(PofReader::Handle)</li>
00033 * </ul>
00034 *
00035 * This class is well suited for use with the coherence::lang::Managed template
00036 * wrapper class. Combining the two allows for the addition of serialization
00037 * support to existing non-coherence classes.
00038 *
00039 * @see coherence::lang::Managed
00040 * @author mf  2008.07.07
00041 */
00042 template<class T>
00043 class TypedSerializer
00044     : public class_spec<TypedSerializer<T>,
00045         extends<Object>,
00046         implements<PofSerializer> >
00047     {
00048     friend class factory<TypedSerializer<T> >;
00049 
00050     // ----- constructors ---------------------------------------------------
00051 
00052     protected:
00053         /**
00054         * Create a new TypedSerializer for the user type with template
00055         * parameter type.
00056         *
00057         * @return the new TypedSerializer
00058         */
00059         TypedSerializer()
00060             {
00061             }
00062 
00063 
00064     // ----- PofSerializer interface ----------------------------------------
00065 
00066     public:
00067         /**
00068         * {@inheritDoc}
00069         */
00070         virtual void serialize(PofWriter::Handle hOut, Object::View v) const
00071             {
00072             ::serialize<typename T::View>(hOut, cast<typename T::View>(v));
00073             hOut->writeRemainder(NULL);
00074             }
00075 
00076         /**
00077         * {@inheritDoc}
00078         */
00079         virtual Object::Holder deserialize(PofReader::Handle hIn) const
00080             {
00081             TypedHolder<T> oh = ::deserialize<TypedHolder<T > >(hIn);
00082             hIn->readRemainder();
00083             return oh;
00084             }
00085     };

00089 /**
00090 * Default serialization implementation which delegates to the serialization
00091 * for a boxed type. This implementation is targeted towards Managed objects
00092 * essentially delegating to serialization functions for the non-managed custom
00093 * type.
00094 *
00095 * @param hOut  the stream to write the object to
00096 * @param t     the object to write out
00097 */
00098 template<class T> void serialize(coherence::io::pof::PofWriter::Handle hOut, const T& t)
00099     {
00100     serialize<typename T::ValueType::BoxedType>(hOut, *t);
00101     }
00102 
00103 /**
00104 * Default deserialization implementation which delegates to the deserialization
00105 * for a boxed type. This implementation is targeted towards Managed objects
00106 * essentially delegating to serialization functions for the non-managed custom
00107 * type.
00108 *
00109 * @param hIn  the stream to read the object from
00110 *
00111 * @return     the deserialized object
00112 */
00113 template<class T> T deserialize(coherence::io::pof::PofReader::Handle hIn)
00114     {
00115     return T::ValueType::create(deserialize<typename T::ValueType::BoxedType>(hIn));
00116     }
