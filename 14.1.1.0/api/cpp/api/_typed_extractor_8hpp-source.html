00024 /**
00025 * Template based ValueExtractor implementation.
00026 *
00027 * This extractor functions on const methods which return Objects, for an
00028 * extractor which works with non-Object return types see BoxExtractor.
00029 *
00030 * For ease of use the COH_TYPED_EXTRACTOR macro can be used to easily construct
00031 * an instance of this class. For example the following constructs an extractor
00032 * for calling the "String::View Address::getState() const" method.
00033 *
00034 * @code
00035 * ValueExtractor::View vExt = COH_TYPED_EXTRACTOR(String::View, Address, getState);
00036 * @endcode
00037 *
00038 * @author mf 2009.03.20
00039 *
00040 * @see BoxExtractor
00041 */
00042 template<class R, class C, R (C::*M)() const, class RH = Object::Holder,
00043     class OH = typename C::Holder>
00044 class TypedExtractor
00045     : public class_spec<TypedExtractor<R, C, M, RH, OH>,
00046         extends<ReflectionExtractor> >
00047     {
00048     friend class factory<TypedExtractor<R, C, M, RH, OH> >;
00049

00058             }

00060         /**
00061         * Construct a TypedExtractor based on a method name and optional
00062         * parameters.
00063         *
00064         * The method name is only used for the purposes of serializing the
00065         * extractor for execution on remote Java members.
00066         *
00067         * @param vsMethod  the name of the method to invoke via reflection
00068         */
00069         TypedExtractor(String::View vsMethod)
00070             : class_spec<TypedExtractor<R, C, M, RH, OH>,
00071             extends<ReflectionExtractor> >(vsMethod)

00073             }

00083             {
00084             // Fix for Bug19826856/COH-12476; strict aliasing failure

00100             }
