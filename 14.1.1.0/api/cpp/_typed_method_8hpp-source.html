00100 /**
00101  * TypedMethod provides a template based implementation for building Methods
00102  * which delegate to member function pointers.
00103  *
00104  * Rather then making direct use of this class it is recommended to utilize
00105  * the following helper macros.  For each of these methods there are nine
00106  * variants suffixed 0..8 which indicate the number of arguments the method
00107  * takes.
00108  *
00109  * - COH_CONSTRUCTOR     defines a Method representing a constructor
00110  * - COH_METHOD          defines a Method representing an instance method
00111  * - COH_STATIC_METHOD   defines a Method representing a static method
00112  * - COH_MANAGED_METHOD  defines a Method representing an instance on a
00113  *                       Managed<> class
00114  *
00115  * - COH_OVERLOADED_CONSTRUCTOR    defines an overloaded constructor
00116  * - COH_OVERLOADED_METHOD         defines an overloaded instance method
00117  * - COH_OVERLOADED_STATIC_METHOD  defines an overloaded static method
00118  *
00119  * Additionally the following macros will create setter and getter Methods
00120  * for a "property".
00121  *
00122  * - COH_PROPERTY          defines setter/getter for a object property
00123  * - COH_BOX_PROPERTY      defines a setter/getter for a value type property
00124  * - COH_MANAGED_PROPERTY  defines a setter/getter for a value type on a
00125  *                         Managed<> class
00126  *
00127  * These macros output Method objects which can then be registered with
00128  * their corresponding Class object.  This registration can either be
00129  * performed as part of the Class registration, or it can be done in multiple
00130  * COH_REGISTER_METHOD blocks, allowing the Method definition to reside
00131  * next to the actual method implementation.
00132  *
00133  * @code
00134  * // Person.hpp
00135  * class Person
00136  *      : public class_spec<Person>
00137  *      {
00138  *      friend class factory<Person>;
00139  *
00140  *      protected:
00141  *          Person(String::View vsName = String::null_string, int32_t nAge = 0);
00142  *
00143  *      public:
00144  *          virtual void setName(String::View vsName);
00145  *          virtual String::View getName() const;
00146  *
00147  *          virtual void setAge(int32_t nAge);
00148  *          virtual int32_t getAge() const;
00149  *      ...
00150  *      };
00151  * @endcode
00152  *
00153  * @code
00154  * // Person.cpp - using single class registration block
00155  *
00156  * COH_REGISTER_CLASS(TypedClass<Person>::create()
00157  *      ->declare(COH_CONSTRUCTOR2(Person, String::View, BoxHandle<const Integer32>))
00158  *      ->declare(COH_PROPERTY(Person, Name, String::View))
00159  *      ->declare(COH_BOX_PROPERTY(Person, Age, Integer32::View));
00160  * @endcode
00161  *
00162  * @code
00163  * // Person.cpp - using multi method registration blocks
00164  *
00165  * COH_REGISTER_TYPED_CLASS(Person);
00166  *
00167  * Person::Person(String::View vsName, int32_t nAge)
00168  *    {
00169  *    ...
00170  *    }
00171  * COH_REGISTER_METHOD(Person, COH_CONSTRUCTOR2(Person, String::View, BoxHandle<const Integer32>));
00172  *
00173  * void Person::setName(String::View vsName)
00174  *     {
00175  *     ...
00176  *     }
00177  * COH_REGISTER_METHOD(Person, COH_METHOD1(void, Person::Handle, setName, String::View));
00178  *
00179  * String::View Person::getName() const
00180  *     {
00181  *     ...
00182  *     }
00183  * COH_REGISTER_METHOD(Person, COH_METHOD0(String::View, Person::View, getName));
00184  *
00185  * void Person::setAge(int32_t nAge)
00186  *     {
00187  *     ...
00188  *     }
00189  * COH_REGISTER_METHOD(Person, COH_METHOD1(void, Person::Handle, setName, BoxHandle<const Integer32>));
00190  *
00191  * int32_t Person::getAge() const
00192  *     {
00193  *     ...
00194  *     }
00195  * COH_REGISTER_METHOD(Person, COH_METHOD0(BoxHandle<const Integer32>, Person::View, getName));
00196  * @endcode
00197  *
00198  * @author mf 2011.02.28
00199  *
00200  * @since Coherence 3.7.1
00201  */
